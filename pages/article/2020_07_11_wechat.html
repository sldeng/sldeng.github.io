# 微信分享

  1. 注意事项
  • 微信分享到朋友圈，朋友；引用JSSDK文件只能在微信浏览器打开才能调用SDK提供的方法
  • 需要在微信公众号后台绑定js安全域名 绑定格式 qyrhdq.cn
  • 需要在微信公众号后台添加ip白名单
  • url要encodeURIComponent
  • 后台需要全局缓存下jsapi_ticket，access_token,微信限制jsapi_ticket，access_token日请求次数，每日限额2000次，且每次获取有效时间为7200s
  1. 代码实现
  init: function(option) {
    // 初始化 微信分享的参数
    console.log(option)
    let url = `${location.href.split('#')[0]}`

    // debugger
    // http://192.168.40.5/
    let link = `${url}#/invite-users?inviteCode=${option.code}`
    console.log('link', link)
    var shareData = {
      title: '一款助农公益新平台', // 分享标题
      desc: '幸福精彩生活由此开启，看短视频也能赚钱。分享给好友，积累新财富！', // 分享描述
      link: link, // 分享链接，该链接域名或路径必须与当前页面对应的公众号JS安全域名一致
      imgUrl: 'http://f.qyrhdq.cn/assets/images/logo-icon.png', // 分享图标
      success: function() {
        // 分享成功可以做相应的数据处理
        //alert("分享成功"); }
      }
    }
    wx.config({
      debug: false, // 开启调试模式,调用的所有api的返回值会在客户端alert出来，若要查看传入的参数，可以在pc端打开，参数信息会通过log打出，仅在pc端时才会打印。
      appId: option.appId, // 必填，公众号的唯一标识
      timestamp:  String(option.timestamp), // 必填，生成签名的时间戳
      nonceStr:  option.nonceStr, // 必填，生成签名的随机串
      signature: option.signature, // 必填，签名
      jsApiList: ['onMenuShareAppMessage', 'onMenuShareTimeline'] // 必填，需要使用的JS接口列表
    })
    wx.checkJsApi({
      jsApiList: ['onMenuShareAppMessage', 'onMenuShareTimeline'], // 需要检测的JS接口列表，所有JS接口列表见附录2,
      success: function(res) {
        // 以键值对的形式返回，可用的api值true，不可用为false
        // 如：{"checkResult":{"chooseImage":true},"errMsg":"checkJsApi:ok"}
        console.log(res)
      },
    })
    wx.ready(function() {
      //需在用户可能点击分享按钮前就先调用 ---分享朋友
      wx.onMenuShareAppMessage(shareData)
      //分享给朋友圈
      wx.onMenuShareTimeline(shareData)
      // wx.updateAppMessageShareData(
      // wx.updateTimelineShareData()
    })
    wx.error(function(res){
      console.log(res,'res')
            // config信息验证失败会执行error函数，如签名过期导致验证失败，具体错误信息可以打开config的debug模式查看，也可以在返回的res参数中查看，对于SPA可以在这里更新签名。
    });
  },

# 微信h5支付

  1. 注意事项
  • 需要在微信商户平台申请h5支付 注意审核时间为3-5个工作日
  • 需要设置h5支付域名
  • 如果需要与app webview中调取需要app设置2. 如果是APP里调起H5支付，需要在webview中手动设置referer，如( Map extraHeaders = new HashMap(); extraHeaders.put("Referer", "商户申请H5时提交的授权域名");//例如 http://www.baidu.com ))
